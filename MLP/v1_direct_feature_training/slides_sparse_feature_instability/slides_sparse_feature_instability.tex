\begin{frame}{Message}
\textbf{Claim:} if sparse operating variables such as nozzle diameter and injection pressure are fed directly into a smooth MLP, the network can invent continuous interpolation behavior that is not supported by the data and is not physically credible.

\vspace{0.5em}
\textbf{How we will show this:}
\begin{itemize}
  \item first show the support topology: where the data manifold is dense, and where it has holes
  \item then compare dense MLP sweeps against secants implied by only the observed sparse levels
  \item finally show that the instability persists over time, especially for nozzle diameter
\end{itemize}
\end{frame}

\begin{frame}{Why This Is a Sparse-Feature Problem}
\textbf{Observed support from the clean audit table:}
\begin{itemize}
  \item injection pressure has only \textbf{4} levels: 1400, 1600, 2000, 2200 bar
  \item nozzle diameter has only \textbf{6} levels: 0.333, 0.348, 0.355, 0.365, 0.375, 0.384 mm
  \item diameter is \emph{not} available uniformly across pressure combinations
  \item all six diameter levels only appear on the narrow family
  \[
  (P_{\mathrm{inj}}=2000,\ C_b=4,\ P_{\mathrm{ch}}\in\{5,10,15\})
  \]
\end{itemize}

\vspace{0.4em}
\textbf{Implication:}
\begin{itemize}
  \item a dense gradient with respect to diameter is mostly an interpolation artifact between a few discrete nozzle levels
  \item even injection-pressure sensitivity is only trustworthy on a small number of supported slices
\end{itemize}
\end{frame}

\begin{frame}{Interpretation of the Support Topology}
\begin{itemize}
  \item left panel: injection pressure support count as a function of chamber pressure and control backpressure
  \item right panels: diameter support count as a function of injection pressure and chamber pressure, split by control backpressure
  \item for most pressure triplets, diameter count is \textbf{1}; the model sees one nozzle level only
  \item therefore a derivative like $\partial \mu / \partial d$ cannot be strongly constrained by data almost anywhere
  \item this is the structural reason the MLP is free to create arbitrary smooth curves between nozzle levels
\end{itemize}

\begin{frame}{Why Diameter Is the Strongest Counterexample}
\textbf{Slide reading guide:}
\begin{itemize}
  \item blue curve: dense MLP sweep over diameter
  \item orange markers: MLP predictions only at actually observed diameter levels
  \item black dashed line: piecewise secant implied by those observed levels only
\end{itemize}

\vspace{0.4em}
\textbf{Observed behavior at $t=\SI{0.85}{ms}$:}
\begin{itemize}
  \item gradient sign changes = \textbf{5--6} across the three supported diameter slices
  \item worst secant-deviation ratio is about \textbf{0.48--0.50}
  \item $p95\ |d^2\mu/dd^2|$ reaches order \textbf{$10^7$}
\end{itemize}

\vspace{0.4em}
\textbf{Interpretation:}
\begin{itemize}
  \item the dense MLP curve bends strongly between sparse nozzle levels
  \item this curvature is not implied by the observed level-to-level trend
  \item physically, this looks like invented interpolation rather than a learned monotonic nozzle-size law
\end{itemize}
\end{frame}

\begin{frame}{Injection Pressure Still Shows Unsupported Behavior}
\textbf{Observed behavior at $t=\SI{0.85}{ms}$:}
\begin{itemize}
  \item the slice \texttt{Pch=15, CB=1, d=0.384} has only \textbf{2} pressure levels
  \item on that slice, worst secant-deviation ratio is \textbf{$>1.0$}
  \item some slices still show \textbf{1--3} gradient sign changes, despite expected positive sensitivity to $P_{\mathrm{inj}}$
\end{itemize}

\vspace{0.4em}
\textbf{Interpretation:}
\begin{itemize}
  \item injection pressure is less pathological than diameter, but the direct-feature MLP still invents smooth structure between sparse pressure levels
  \item on weakly supported slices, the continuous derivative field becomes a modeling choice, not a data-driven conclusion
\end{itemize}
\end{frame}

\begin{frame}{Time-Series Interpretation}
\begin{itemize}
  \item diameter instability is persistent, not a single-time anomaly
  \item across time, diameter keeps large secant-deviation ratios, many sign changes, and very large curvature
  \item injection pressure is more slice-dependent: some families remain relatively smooth, while others become highly unreliable when support is thin
  \item this means the issue is not just local optimization noise; it is a systematic consequence of sparse-feature geometry in the training set
\end{itemize}
\end{frame}

\begin{frame}{Engineering Takeaway}
\textbf{What these slides support:}
\begin{itemize}
  \item direct continuous learning on sparse nozzle diameter is not trustworthy for physical sensitivity analysis
  \item direct continuous learning on sparse injection pressure can also fail on weakly supported slices
  \item an MLP can fit training loss while still producing anti-physical derivative structure between sparse feature levels
\end{itemize}

\vspace{0.4em}
\textbf{Practical next steps:}
\begin{itemize}
  \item treat diameter as a discrete nozzle identity or hierarchical effect, not a free continuous variable
  \item constrain sensitivity using monotonic priors / secant regularization / family-wise models
  \item always pair dense sweeps with support maps before interpreting gradients physically
\end{itemize}
