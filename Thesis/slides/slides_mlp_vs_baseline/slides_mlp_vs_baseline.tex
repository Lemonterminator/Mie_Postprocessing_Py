\begin{frame}{Scope of This Overview}

This short deck is the bridge between method families. The detailed evidence is
split into three thesis decks: baseline comparison, engineered-feature scaling,
and probabilistic calibration.
\end{block}

\begin{block}{Current headline}
On the common external clean diagnostic set
(\num{692942} points, \num{24316} trajectories), the production Stage-3 MLP
five-seed mean strongly beats calibrated Hiroyasu--Arai and Naber--Siebers.
The Stage-3 SVGP remains the point-accuracy winner; the MLP is the more
conservative screening surrogate.
\end{block}

\begin{frame}{Traditional Baseline I: Naber--Siebers Correlation}
\textbf{Formulation:}
The Naber--Siebers model correlates spray penetration $S(t)$ with physical parameters:
\[
S(t) = K \cdot \underbrace{\left( \frac{\Delta P}{\rho_a} \right)^{0.25} \sqrt{d_n}}_{\text{Scaling Factor } A} \cdot \sqrt{\max(t - t_d, 0)}
\]
\vspace{0.2em}
\textbf{Fitted Parameters:}
\begin{itemize}
    \item $K$: Global scaling constant.
    \item $t_d$: Fitted temporal delay (accounting for hydraulic lag).
\end{itemize}
\vspace{0.2em}
\textbf{Image Processing Data Utilized:}
\begin{itemize}
    \item \textbf{CDF Penetration Series:} Cleaned, delay-aligned penetration trajectories from top-view Mie scattering.
    \item \textbf{Cone Angle $\theta$ (Optional Variant):} Incorporates spreading factor $\frac{1}{\sqrt{\max(\tan(\theta/2), 10^{-9})}}$ extracted directly from video segmentation.
\end{itemize}
\end{frame}

\begin{frame}{Traditional Baseline II: Hiroyasu--Arai (1990) \& Zhou (2019)}
\textbf{Formulation (Multi-stage):}
\begin{itemize}
    \item \textbf{Early Stage ($t < t_b$):} $S = k_v \sqrt{\frac{2 \Delta P}{\rho_l}} t$
    \item \textbf{Late Stage ($t \geq t_b$):} $S = k_p \left( \frac{\Delta P}{\rho_a} \right)^{0.25} \sqrt{d_n t}$
    \item \textbf{Breakup Time:} $t_b = k_{bt} \frac{\rho_l d_n}{\sqrt{\rho_a \Delta P}}$
    \item \textbf{Zhou Extension (Post-injection, $t > 2t_i$):}
    \[ S = \sqrt{2} k_p \left( \frac{\Delta P}{\rho_a} \right)^{0.25} \sqrt{d_n} t_i^{0.25} (t - t_i)^{0.25} \]
\end{itemize}

\vspace{0.2em}
\textbf{Fitted Parameters:}
\begin{itemize}
    \item $k_v, k_p, k_{bt}$ calibrated on the dataset (or literature defaults).
\end{itemize}
\end{frame}

\begin{frame}{GP Learning Target: Same Regression, Different Prior}
\small
\begin{block}{Start from the same supervised-learning problem as the MLP}
\footnotesize
Each pointwise training sample is
\[
x_i=[t,\;\mathrm{tilt},\;\mathrm{plumes},\;t_{\mathrm{inj}},\;p_{\mathrm{ctrl}}]_i,
\qquad
z_i=\frac{S_i}{A_i}.
\]
The physical scaling factor is
\[
A_i=(\Delta P_i)^{0.5}\rho_{a,i}^{-0.25}d_{n,i}^{0.5}.
\]
The MLP learns \(x\mapsto z\). The GP baseline learns the same target and is
multiplied by \(A_i\) at the end to return to millimetres.
\end{block}

\begin{block}{The difference is the function assumption}
\footnotesize
MLP: represent the function explicitly with parameters, \(z_\theta(x)\).\\
GP: place a probability distribution directly over the unknown function:
\[
z_i=f(x_i)+\epsilon_i.
\]
\end{block}
\end{frame}

\begin{frame}{GP Definition: A Distribution over Functions}
\small
\begin{block}{Finite-dimensional definition}
\footnotesize
For any finite input set \(X=[x_1,\ldots,x_n]\), if the function values satisfy
\[
\mathbf f_X=[f(x_1),\ldots,f(x_n)]^\top
\sim
\mathcal N(\mathbf m_X,K_{XX}),
\]
then
\[
f(x)\sim\mathcal{GP}(m(x),k(x,x')).
\]
\end{block}

\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\linewidth}
\begin{block}{Mean function}
\footnotesize
\[
(\mathbf m_X)_i=m(x_i)
\]
The implementation uses a constant mean, so the data learn the global offset.
\end{block}
\end{column}
\begin{column}{0.48\linewidth}
\begin{block}{Kernel / covariance}
\footnotesize
\[
(K_{XX})_{ij}=k(x_i,x_j)
\]
The kernel says how similar two function values should be.
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Kernel Used Here: ARD Matérn-\(5/2\)}
\small
\begin{block}{Distance is scaled by learned length-scales}
\footnotesize
\[
r^2(x,x')=\sum_{j=1}^{5}\left(\frac{x_j-x'_j}{\ell_j}\right)^2.
\]
\(\ell_j\) is the length-scale for input dimension \(j\). A large \(\ell_j\)
means the function is insensitive to that feature; a small \(\ell_j\) means
small changes in that feature can change the prediction.
\end{block}

\begin{block}{Matérn-\(5/2\) covariance}
\footnotesize
\[
k_\theta(x,x')=\sigma_f^2
\left(1+\sqrt{5}r+\frac{5r^2}{3}\right)\exp(-\sqrt{5}r).
\]
It is smooth, but less over-smooth than an RBF kernel, which is useful for
penetration traces with transient behaviour.
\end{block}

\vspace{-0.2em}
\footnotesize
Code: \texttt{ScaleKernel(MaternKernel(nu=2.5, ard\_num\_dims=5))}.
\end{frame}

\begin{frame}{Exact GP Posterior: Joint Gaussian Conditioning}
\small
\begin{block}{Start with the homoscedastic observation model}
\footnotesize
\[
\mathbf z=\mathbf f_X+\boldsymbol\epsilon,
\qquad
\boldsymbol\epsilon\sim\mathcal N(0,\sigma_n^2I).
\]
For test inputs \(X_*\), the joint distribution is
\[
\begin{bmatrix}
\mathbf z\\
\mathbf f_*
\end{bmatrix}
\sim
\mathcal N\left(
\begin{bmatrix}
\mathbf m_X\\
\mathbf m_*
\end{bmatrix},
\begin{bmatrix}
K_{XX}+\sigma_n^2I & K_{X*}\\
K_{*X} & K_{**}
\end{bmatrix}
\right).
\]
\end{block}

\begin{block}{Conditioning gives the prediction formula}
\footnotesize
\[
\mu_*=\mathbf m_*+K_{*X}(K_{XX}+\sigma_n^2I)^{-1}(\mathbf z-\mathbf m_X),
\]
\[
\Sigma_*=K_{**}-K_{*X}(K_{XX}+\sigma_n^2I)^{-1}K_{X*}.
\]
This is the mathematical form of ``average nearby training evidence and return
an uncertainty''.
\end{block}
\end{frame}

\begin{frame}{Why Exact GP Is Not Feasible Here}
\small
\begin{block}{Hyperparameters are learned by marginal likelihood}
\footnotesize
Let \(K_y=K_{XX}+\sigma_n^2I\). Then
\[
\log p(\mathbf z|X,\theta)
=-\frac{1}{2}(\mathbf z-\mathbf m)^\top K_y^{-1}(\mathbf z-\mathbf m)
-\frac{1}{2}\log|K_y|-\frac{n}{2}\log 2\pi.
\]
The first term rewards fit, the second penalizes overly flexible covariance,
and the third is constant.
\end{block}

\begin{block}{The scaling bottleneck}
\footnotesize
Computing \(K_y^{-1}\mathbf z\) and \(\log|K_y|\) usually requires a Cholesky
factorization:
\[
K_y=LL^\top,\qquad \mathrm{cost}=O(n^3),\quad \mathrm{memory}=O(n^2).
\]
This task expands trajectories into many time points; per-seed training can
reach the million-point scale. Exact GP is therefore not practical.
\end{block}
\end{frame}

\begin{frame}{Sparse GP: Compress the Function with Inducing Points}
\small
\begin{block}{Core idea}
\footnotesize
Choose \(m\ll n\) inducing inputs
\[
Z=[z_1^{(u)},\ldots,z_m^{(u)}],
\qquad
\mathbf u=f(Z).
\]
They are not labels. They are support locations for the latent function.
Training learns the function distribution around these support locations and
uses them to explain all \(n\) observations.
\end{block}

\begin{block}{Conditional structure}
\footnotesize
The GP prior gives
\[
p(\mathbf f,\mathbf u)=p(\mathbf f|\mathbf u)p(\mathbf u),
\qquad
p(\mathbf u)=\mathcal N(\mathbf m_Z,K_{ZZ}).
\]
With \(m=256\), the main operations are built around \(m\times m\) matrices and
minibatch covariance blocks instead of one dense \(n\times n\) matrix.
\end{block}

\footnotesize
In this implementation, the 256 inducing points are initialized by K-means on
the training inputs and then allowed to move during training.
\end{frame}

\begin{frame}{Variational GP: Optimize \(q(u)\) Instead of Exact \(p(u|D)\)}
\small
\begin{block}{Variational distribution}
\footnotesize
The exact posterior \(p(\mathbf u|\mathcal D)\) is hard to compute, so we use
\[
q(\mathbf u)=\mathcal N(\mathbf m_u,S_u),
\]
with \(\mathbf m_u\) and \(S_u\) as trainable parameters. For any training
function values,
\[
q(\mathbf f)=\int p(\mathbf f|\mathbf u)q(\mathbf u)\,d\mathbf u.
\]
\end{block}

\begin{block}{ELBO}
\footnotesize
\[
\mathcal L
=
\sum_{i=1}^{n}
\mathbb E_{q(f_i)}[\log p(z_i|f_i)]
-
\mathrm{KL}\!\left(q(\mathbf u)\,\|\,p(\mathbf u)\right).
\]
The first term explains the data; the second prevents \(q(u)\) from drifting
too far from the GP prior. In minibatch training, the likelihood part is scaled
by \(n/|\mathcal B|\).
\end{block}
\end{frame}

\begin{frame}{The Thesis GP Variant: Two SVGPs in Series}
\small
\begin{block}{Stage 1: mean SVGP}
\footnotesize
\[
f_\mu(x)\sim\mathcal{GP}(m_\mu,k_\mu),
\qquad
z_i=f_\mu(x_i)+\epsilon_i.
\]
After training, it gives
\[
\hat\mu_z(x)=\mathbb E[f_\mu(x)].
\]
\end{block}

\begin{block}{Stage 2: log-variance SVGP}
\footnotesize
First compute residuals from the mean GP:
\[
r_i=z_i-\hat\mu_z(x_i).
\]
Then construct the variance target:
\[
g_i=\log(r_i^2+\varepsilon).
\]
A second SVGP learns
\[
f_v(x)\approx g,
\qquad
\hat\sigma_z^2(x)=\exp(\hat f_v(x)+b).
\]
\(b\) is a validation-set log-variance bias correction.
\end{block}
\end{frame}

\begin{frame}{Returning from Scaled GP Space to Millimetres}
\footnotesize
\begin{block}{Scaled space}
\scriptsize
The GP is trained and evaluated internally on
\[
z=\frac{S}{A_{\mathrm{scale}}},
\]
so it predicts
\[
\hat\mu_z(x),\qquad \hat\sigma_z(x).
\]
\end{block}

\begin{block}{Physical space}
\scriptsize
For penetration in millimetres:
\[
\hat\mu_S(x)=A_{\mathrm{scale}}(x)\hat\mu_z(x),
\qquad
\hat\sigma_S(x)=A_{\mathrm{scale}}(x)\hat\sigma_z(x).
\]
RMSE, MAE, P95, and coverage are all computed in this physical space.
\end{block}

\vspace{-0.2em}
\scriptsize
\textbf{Implementation detail.}
The headline GP uses residual log-variance as the uncertainty source:
\texttt{include\_mean\_posterior\_var=false}, so reported \(\sigma\) does not
add the mean GP posterior variance.
\end{frame}

\begin{frame}{Why the GP Baseline Is Fair, but Still Loses on Accuracy}
\small
\begin{block}{Why it is fair}
\footnotesize
The Stage-3 SVGP and production MLP both consume the same full-data information
class: synthetic Q1 supervision plus real clean CDF observations. They are then
scored on the same full-clean and fixed-table evaluation sets.
\end{block}

\begin{block}{Where GP is strong}
\footnotesize
The kernel prior gives a strong smoothness bias, SVGP gives probabilistic
predictions, and the second log-variance GP makes uncertainty input-dependent.
It is not a weak baseline; it brings coverage close to nominal.
\end{block}

\begin{block}{Where the MLP pipeline is stronger}
\footnotesize
The Stage-3 MLP is not just a generic regressor. It inherits the full fitted
trajectory target structure and then refines early-time behaviour against raw
CDF observations. The GP learns the same pointwise target family, but it does
not include that teacher/raw staged training mechanism. In the current results,
this advantage appears mainly as conservative coverage rather than lower RMSE.
\end{block}
\end{frame}

\begin{frame}{MLP vs. Empirical Correlations}
\textbf{Why Compare against these Baselines?}
\begin{itemize}
    \item \textbf{Rigidity of Traditional Models:} Empirical formulas lock the penetration curve into $\sqrt{t}$ or $t^{0.25}$ dependencies, ignoring specific internal nozzle geometry or complex post-injection pressure wave dynamics.
    \item \textbf{Advantage of MLP:} With the \textbf{upgraded engineered features} (e.g., $A=\Delta P^{0.5}\rho_a^{-0.25}d_n^{0.5}$), the MLP is anchored to physical dimensions but retains the capacity to learn non-linear residuals.
    \item \textbf{Data Utilization:} Naber-Siebers optional variant relies on the explicit video-measured cone angle $\theta$. The MLP aims to predict penetration mapping directly from controllable condition parameters, without needing "oracle" visual features during inference.
\end{itemize}
\end{frame}
