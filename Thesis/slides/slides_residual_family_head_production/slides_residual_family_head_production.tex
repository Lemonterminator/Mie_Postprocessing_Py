\begin{frame}{Message}
\small
\textbf{Claim:} converting the deployed family-aware Stage-3 model into a
\emph{residual family head}
\[
  \mu(x,f) = \mu_{\rm shared}(h(x)) + \Delta_f(h(x))
\]
improves the production point-table metrics while preserving the deployment
path for new injector families.

\vspace{0.5em}
\textbf{Best run:} \texttt{delta\_L2 = 1e-4}.\\
\textbf{Primary metric:} CDF uncensored RMSE improves
\good{5.056 $\to$ 4.631 mm}.\\
\textbf{Secondary metrics:} P50 RMSE improves
\good{3.931 $\to$ 3.395 mm}; CDF ECE improves
\good{0.0708 $\to$ 0.0604}.

\begin{frame}{Architecture: Residual Family Head}
\small
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{Current family head}
\[
  \mu(x,f) = \mu_f(h(x))
\]
Each family owns the full mean decoder. A new family needs a fallback policy or
a trained head.

\vspace{0.5em}
\textbf{Residual family head}
\[
  \mu(x,f) = \mu_{\rm shared}(h(x)) + \Delta_f(h(x))
\]
The shared mean is copied from the modified-family production head
(\texttt{mu\_heads.1}); each residual head is zero-initialised.

\column{0.44\textwidth}
\begin{block}{Warm-start mapping}
\begin{itemize}\setlength\itemsep{2pt}
  \item copy \texttt{trunk.*}
  \item copy \texttt{log\_var\_head.*}
  \item \texttt{mu\_heads.1} $\to$ \texttt{mu\_shared\_head}
  \item \texttt{delta\_heads.*}: zero init, then KD mimic
\end{itemize}
\end{block}

\begin{block}{Deployment property}
An unseen family with zero residual naturally predicts
\(\mu_{\rm shared}\); no hard-coded \texttt{fallback\_family\_id} is needed.
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Verdict}
\small
\begin{enumerate}\setlength\itemsep{4pt}
  \item The deployed production model can be hot-started into a residual family
        head without retraining the trunk.
  \item The best residual model beats current production on the primary metric:
        CDF uncensored RMSE \good{5.056 $\to$ 4.631 mm}.
  \item P50 and observed-window Q1 also improve substantially:
        P50 \good{3.931 $\to$ 3.395 mm}; Q1 observed
        \good{3.839 $\to$ 3.242 mm}.
  \item Extrapolated Q1 is effectively flat for the winner
        (11.295 $\to$ 11.307 mm); stronger shrinkage improves this tail
        slightly at a small CDF-RMSE cost.
  \item Next step: package the residual winner as the new candidate deployed
        model, then run a seed check or a small new-family calibration simulation.
\end{enumerate}

\begin{frame}{Next MLP-Only Ablation}
\small
\begin{block}{Recommended architecture: residual FiLM adapter}
Move the family signal from only the output head into the hidden
representation while preserving exact hot-start compatibility:
\[
  h_l' = \gamma_{g,l}\odot h_l + \beta_{g,l}, \qquad
  \mu(x,g)=\mu_{\rm shared}(h_L')+\Delta_g(h_L')
\]
\end{block}

\vspace{0.2em}
\begin{itemize}\setlength\itemsep{3pt}
  \item Identity init: \(\gamma_{g,l}=1\), \(\beta_{g,l}=0\), and
        \(\Delta_g=0\), so the copied production model is unchanged at step 0.
  \item Unknown family fallback: identity adapter plus zero residual gives the
        shared predictor without a hard-coded fallback family.
  \item MLP-specific question: does injector family change the representation
        of \texttt{a\_dp050\_plus\_pressures}, not just the final mean offset?
\end{itemize}
\end{frame}
